# Implementazione

- **Time:** $O(N\log N)$
- **Memory:** $O(N)$

```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
	    dict = collections.defaultdict(list)

		for str in strs:
			key = ''.join(sorted(str))
			dict[key].append(str)
		
		return dict.values()
```

### Mia Soluzione
```python
class Solution:
    def groupAnagrams(self, strs: List[str]) -> List[List[str]]:
        res = []
        n = len(strs)
        if n == 1:
            return [strs]

        seen = [False] * (n+1)        

        for i in range(n):
            if seen[i]:
                continue

            need = {}
            for c in strs[i]:
                need[c] = need.get(c, 0) + 1

            seen[i] = True
            curr = [strs[i]]

            for j in range(i+1, n):
                if seen[j]:
                    continue
                
                have = {}
                isAnagram = True
                for c in strs[j]:
                    have[c] = have.get(c, 0) + 1
                    if have[c] > need.get(c, 0):
                        isAnagram = False
                
                if isAnagram:
                    for key, item in need.items():
                        if have.get(key, 0) != item:
                            isAnagram = False
                            break
                    if isAnagram:
                        seen[j] = True
                        curr.append(strs[j])

            res.append(curr)
        
        return res
```
